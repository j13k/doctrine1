  180.         $query = '';

  190.         if ( ! empty($changes['remove']) && is_array($changes['remove'])) {
  191.             foreach ($changes['remove'] as $fieldName => $field) {

  200.         if ( ! $query) {

  205.         return $this->conn->exec('ALTER TABLE ' . $name . ' ' . $query);
