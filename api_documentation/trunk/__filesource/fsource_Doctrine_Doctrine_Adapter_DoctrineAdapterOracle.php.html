  118.             $this->_connection = @oci_connect(
  119.                 $this->_config['username'],
  120.                 $this->_config['password'],
  121.                 $this->_config['dbname']);
  122.         } else {

  124.                 $this->_config['username'],
  125.                 $this->_config['password']);
  126.         }
  127.  
  128.         // check the connection
  129.         if (!$this->_connection) {

  221.     /**

  230.     public function setFetchMode($mode)

  236.             case Doctrine::FETCH_OBJ:   // object
  237.                 $this->_fetchMode = $mode;
  238.                 break;
  239.             default:
  240.                 throw new Doctrine_Adapter_Exception('Invalid fetch mode specified');
  241.                 break;
  242.         }
  243.     }
  244.     /**

  248.     private function _setExecuteMode($mode)

  254.                 $this->_execute_mode = $mode;
  255.                 break;
  256.             default:

  258.                 break;
  259.         }
  260.     }
  261.     /**
