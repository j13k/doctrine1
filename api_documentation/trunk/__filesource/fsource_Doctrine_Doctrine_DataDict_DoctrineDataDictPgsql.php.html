  338.     /**

  345.      *
  346.      *       length
  347.      *           Integer value that determines the maximum length of the text
  348.      *           field. If this argument is missing the field should be
  349.      *           declared to have the longest length allowed by the DBMS.
  350.      *
  351.      *       default

  353.      *
  354.      *       notnull

  356.      *           to not be set to null.
  357.      *
  358.      * @return string  DBMS specific SQL code portion that should be used to
  359.      *       declare the specified field.
  360.      */
  361.     public function getNativeDeclaration(array $field)
  362.     {
  363.         if ( ! isset($field['type'])) {
  364.             throw new Doctrine_DataDict_Exception('Missing column type.');
  365.         }
  366.         switch ($field['type']) {
  367.             case 'char':
  368.             case 'string':
  369.             case 'array':
  370.             case 'object':
  371.             case 'varchar':   
  372.             case 'gzip':
  373.                 $length = (isset($field['length']) && $field['length']) ? $field['length'] : null;
  374.                         // TODO:  $this->conn->options['default_text_field_length'];
  375.  
  376.                 $fixed  = ((isset($field['fixed']) && $field['fixed']) || $field['type'] == 'char') ? true : false;
  377.  
  378.                 return $fixed ? ($length ? 'CHAR('.$length.')' : 'CHAR('.$this->conn->options['default_text_field_length'].')')
  379.                     : ($length ? 'VARCHAR('.$length.')' : 'TEXT');
  380.  
  381.             case 'clob':
  382.                 return 'TEXT';
  383.             case 'blob':
  384.                 return 'BYTEA';
  385.             case 'enum':
  386.             case 'integer':
  387.             case 'int':
  388.                 if (!empty($field['autoincrement'])) {
  389.                     if (!empty($field['length'])) {
  390.                         $length = $field['length'];
  391.                         if ($length > 4) {
  392.                             return 'BIGSERIAL';
  393.                         }
  394.                     }
  395.                     return 'SERIAL';
  396.                 }
  397.                 if (!empty($field['length'])) {
  398.                     $length = $field['length'];
  399.                     if ($length <= 2) {
  400.                         return 'SMALLINT';
  401.                     } elseif ($length == 3 || $length == 4) {
  402.                         return 'INT';
  403.                     } elseif ($length > 4) {
  404.                         return 'BIGINT';
  405.                     }
  406.                 }
  407.                 return 'INT';
  408.             case 'boolean':
  409.                 return 'BOOLEAN';
  410.             case 'date':
  411.                 return 'DATE';
  412.             case 'time':
  413.                 return 'TIME without time zone';
  414.             case 'timestamp':
  415.                 return 'TIMESTAMP without time zone';
  416.             case 'float':
  417.             case 'double':
  418.                 return 'FLOAT8';
  419.             case 'decimal':
  420.                 $length = !empty($field['length']) ? $field['length'] : 18;
  421.                 $scale = !empty($field['scale']) ? $field['scale'] : $this->conn->getAttribute(Doctrine::ATTR_DECIMAL_PLACES);
  422.                 return 'NUMERIC('.$length.','.$scale.')';
  423.         }
  424.         throw new Doctrine_DataDict_Exception('Unknown field type \'' . $field['type'] .  '\'.');
  425.     }

  465.             case 'int':
  466.             case 'int4':

  468.             case 'serial':
  469.             case 'serial4':

  471.                 $unsigned = false;
  472.                 $length = 4;
  473.                 break;
  474.             case 'bigint':
  475.             case 'int8':
  476.             case 'bigserial':
  477.             case 'serial8':
  478.                 $type[] = 'integer';
  479.                 $unsigned = false;
  480.                 $length = 8;
  481.                 break;
  482.             case 'bool':
  483.             case 'boolean':
  484.                 $type[] = 'boolean';
  485.                 $length = 1;
  486.                 break;
  487.             case 'text':
  488.             case 'varchar':
  489.                 $fixed = false;
  490.             case 'unknown':
  491.             case 'char':
  492.             case 'bpchar':
  493.                 $type[] = 'string';
  494.                 if ($length == '1') {
  495.                     $type[] = 'boolean';
  496.                     if (preg_match('/^(is|has)/', $field['name'])) {
  497.                         $type = array_reverse($type);
  498.                     }
  499.                 } elseif (strstr($dbType, 'text')) {
  500.                     $type[] = 'clob';
  501.                 }
  502.                 if ($fixed !== false) {
  503.                     $fixed = true;
  504.                 }
  505.                 break;
  506.             case 'date':
  507.                 $type[] = 'date';
  508.                 $length = null;
  509.                 break;
