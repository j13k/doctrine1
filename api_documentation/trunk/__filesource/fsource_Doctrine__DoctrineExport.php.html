  570.     /**

  644.  
  645.         $charset   = (isset($field['charset']) && $field['charset']) ?
  646.                     ' ' . $this->getCharsetFieldDeclaration($field['charset']) : '';
  647.  
  648.         $collation = (isset($field['collation']) && $field['collation']) ?
  649.                     ' ' . $this->getCollationFieldDeclaration($field['collation']) : '';

  652.  
  653.         $unique    = (isset($field['unique']) && $field['unique']) ?
  654.                     ' ' . $this->getUniqueFieldDeclaration() : '';
  655.                     
  656.         $check     = (isset($field['check']) && $field['check']) ?
  657.                     ' ' . $field['check'] : '';

  660.  
  661.         if (method_exists($this->conn->dataDict, $method)) {

  663.         } else {
  664.             $dec = $this->conn->dataDict->getNativeDeclaration($field);
  665.         }

  668.     /**
