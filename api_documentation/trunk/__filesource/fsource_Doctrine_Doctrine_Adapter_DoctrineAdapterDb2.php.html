  107.         if ( ! isset($config['dbname'])) {

  126.      *

  129.     protected function _connect()
  130.     {
  131.         if (is_resource($this->_connection)) {
  132.             // connection already exists
  133.             return;
  134.         }
  135.  
  136.         if ( ! extension_loaded('ibm_db2')) {
  137.             throw new Doctrine_Adapter_Db2_Exception('The IBM DB2 extension is required for this adapter but not loaded');
  138.         }
  139.  
  140.         if ($this->_config['persistent']) {
  141.             // use persistent connection
  142.             $conn_func_name = 'db2_pconnect';
  143.         } else {
  144.             // use "normal" connection
  145.             $conn_func_name = 'db2_connect';
  146.         }
  147.  
  148.         if (!isset($this->_config['options'])) {
  149.             // config options were not set, so set it to an empty array
  150.             $this->_config['options'] = array();
  151.         }
  152.  
  153.         if (!isset($this->_config['options']['autocommit'])) {
  154.             // set execution mode
  155.             $this->_config['options']['autocommit'] = &$this->_execute_mode;
  156.         }
  157.  
  158.         if ($this->_config['host'] !== 'localhost') {
  159.             // if the host isn't localhost, use extended connection params
  160.             $dbname = 'DRIVER={IBM DB2 ODBC DRIVER}' .
  161.                      ';DATABASE=' . $this->_config['dbname'] .
  162.                      ';HOSTNAME=' . $this->_config['host'] .
  163.                      ';PORT='     . $this->_config['port'] .
  164.                      ';PROTOCOL=' . $this->_config['protocol'] .
  165.                      ';UID='      . $this->_config['username'] .
  166.                      ';PWD='      . $this->_config['password'] .';';
  167.             $this->_connection = $conn_func_name(
  168.                 $dbname,
  169.                 null,
  170.                 null,
  171.                 $this->_config['options']
  172.             );
  173.         } else {
  174.             // host is localhost, so use standard connection params
  175.             $this->_connection = $conn_func_name(
  176.                 $this->_config['dbname'],
  177.                 $this->_config['username'],
  178.                 $this->_config['password'],
  179.                 $this->_config['options']
  180.             );
  181.         }
  182.  
  183.         // check the connection
  184.         if (!$this->_connection) {
  185.             throw new Doctrine_Adapter_Db2_Exception(db2_conn_errormsg(), db2_conn_error());
  186.         }
  187.     }

  328.             default:

  330.                 break;
  331.         }
