  199.             }
  200.  
  201.             $driverName = $adapter->getAttribute(Doctrine::ATTR_DRIVER_NAME);
  202.         } elseif (is_array($adapter)) {

  204.                 throw new Doctrine_Manager_Exception('Empty data source name given.');
  205.             }

  207.  
  208.             if($e[0] == 'uri') {
  209.                 $e[0] = 'odbc';
  210.             }
  211.  
  212.             $parts['dsn']    = $adapter[0];
  213.             $parts['scheme'] = $e[0];
  214.             $parts['user']   = (isset($adapter[1])) ? $adapter[1] : null;
  215.             $parts['pass']   = (isset($adapter[2])) ? $adapter[2] : null;
  216.             
  217.             $driverName = $e[0];
  218.             $adapter = $parts;
  219.         } else {
  220.             $parts = $this->parseDsn($adapter);
  221.             
  222.             $driverName = $parts['scheme'];
  223.             
  224.             $adapter = $parts;

  226.

  230.         if ($name !== null) {

  233.                 return $this->_connections[$name];

  264.     /**

  277.         foreach ($names as $name) {

  283.         if (count($parts) == 0 || ! isset($parts['scheme'])) {
  284.             throw new Doctrine_Manager_Exception('Empty data source name');
  285.         }
  286.  
  287.         switch ($parts['scheme']) {
  288.             case 'sqlite':
  289.             case 'sqlite2':

  291.                 if (isset($parts['host']) && $parts['host'] == ':memory') {
  292.                     $parts['database'] = ':memory:';

  294.                 } else {
  295.                     $parts['database'] = $parts['path'];
  296.                     $parts['dsn'] = $parts['scheme'] . ':' . $parts['path'];         
  297.                 }
  298.  
  299.                 break;
  300.             case 'mysql':
  301.             case 'informix':
  302.             case 'oci8':
  303.             case 'oci':
  304.             case 'mssql':
  305.             case 'firebird':
  306.             case 'dblib':
  307.             case 'pgsql':
  308.             case 'odbc':
  309.             case 'mock':
  310.             case 'oracle':
  311.                 if ( ! isset($parts['path']) || $parts['path'] == '/') {
  312.                     throw new Doctrine_Manager_Exception('No database available in data source name');
  313.                 }
  314.                 if (isset($parts['path'])) {

  316.                 }
  317.                 if ( ! isset($parts['host'])) {
  318.                     throw new Doctrine_Manager_Exception('No hostname set in data source name');
  319.                 }
  320.                 
  321.                 if (isset(self::$driverMap[$parts['scheme']])) {
  322.                     $parts['scheme'] = self::$driverMap[$parts['scheme']];
  323.                 }
  324.  
  325.                 $parts['dsn'] = $parts['scheme'] . ':host='
  326.                               . $parts['host'] . ';dbname='
  327.                               . $parts['database'];
  328.                 
  329.                 if (isset($parts['port'])) {
  330.                     // append port to dsn if supplied
  331.                     $parts['dsn'] .= ';port=' . $parts['port'];
  332.                 }
  333.                 break;
  334.             default:
  335.                 throw new Doctrine_Manager_Exception('Unknown driver '.$parts['scheme']);
  336.         }
  337.  
  338.  
  339.         return $parts;
  340.     }
